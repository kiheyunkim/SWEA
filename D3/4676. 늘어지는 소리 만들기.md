```c
//C
#include <stdio.h>
#include <string.h>
int count[20 + 2 + 1];
char inputString[20 + 1];
int main(void) {
	int T;
	scanf("%d", &T);
	for (int test_case = 1; test_case <= T; ++test_case) {
		memset(count, 0, sizeof(count));
		int h;
		scanf("%s", inputString);
		int length = strlen(inputString);
		scanf("%d", &h);
		for (int i = 0; i < h; ++i) {
			int input;
			scanf("%d", &input);
			++count[input];
		}

		printf("#%d ", test_case);
		for (int i = 0; i < length + 2; ++i) {
			for (int j = 0; j < count[i]; ++j) {
				printf("%c", '-');
			}

			if (0 <= i && i < length) {
				printf("%c", inputString[i]);
			}
		}
		printf("\n");
	}

	return 0; //정상종료시 반드시 0을 리턴해야 합니다.
}
```

```java
//java
import java.util.Scanner;
import java.io.FileInputStream;

class Solution {
    public static void main(String args[]) throws Exception {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        for (int test_case = 1; test_case <= T; test_case++) {
            String input = sc.next();
            int inputLen = input.length();
            int[] counter = new int[inputLen + 2];
            int h = sc.nextInt();
            for (int i = 0; i < h; ++i) {
                int number = sc.nextInt();
                ++counter[number];
            }

            System.out.print(String.format("#%d ", test_case));
            for (int i = 0; i < inputLen + 2; ++i) {
                for (int j = 0; j < counter[i]; ++j) {
                    System.out.print('-');
                }

                if (i < inputLen) {
                    System.out.print(input.charAt(i));
                }
            }
            System.out.println();

        }
    }
}
```

```python
T = int(input())
for test_case in range(1, T + 1):
    inputString = str(input())
    inputLength = int(len(inputString))
    h = int(input())
    numbers = list(map(int, input().split(" ")))

    counter = [0 for _ in range(0, inputLength + 2)]
    for i in range(0, h):
        counter[numbers[i]] += 1

    print("#{0} ".format(test_case), end="")
    for i in range(0, inputLength + 2):
        for j in range(0, counter[i]):
            print('-', end="")
        if i < inputLength:
            print(inputString[i], end='')
    print()
```
