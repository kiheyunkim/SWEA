//C
#include <stdio.h>
int stack[100000 + 1];
int stackSize = 0;
int main(void){
	int test_case;
	int T, K, i, input,answer;
	
	setbuf(stdout, NULL);
	scanf("%d", &T);
	for (test_case = 1; test_case <= T; ++test_case) {
		answer = 0;
		scanf("%d", &K);
		for (i = 0; i < K; ++i) {
			scanf("%d", &input);
			if (input == 0) {
				--stackSize;
			}
			else {
				stack[stackSize++] = input;
			}
		}

		while (stackSize >0) {
			stackSize -= 1;
			answer += stack[stackSize];
		}

		printf("#%d %d\n", test_case, answer);
	}

	return 0; //정상종료시 반드시 0을 리턴해야 합니다.
}

//C++
#include<iostream>
#include<stack>
int main(int argc, char** argv){
	std::cin.tie(nullptr);
	std::cout.tie(nullptr);
	std::ios::sync_with_stdio(false);

	int T;
	std::cin >> T;
	for (int test_case = 1; test_case <= T; ++test_case){
		std::stack<int> stack;
		int K, input;
		std::cin >> K;

		for (int i = 0; i < K; ++i) {
			std::cin >> input;
			if (input == 0) {
				stack.pop();
			}
			else {
				stack.push(input);
			}
		}

		int answer = 0;
		while (!stack.empty()){
			answer += stack.top();
			stack.pop();
		}

		std::cout << "#" << test_case << " " << answer << "\n";
	}
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}

import java.util.Scanner;
import java.io.FileInputStream;

class Solution {
    public static void main(String args[]) throws Exception {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        for (int test_case = 1; test_case <= T; test_case++) {
            int K = sc.nextInt();
            int[] stack = new int[100000 + 1];
            int stackSize = 0;

            for (int i = 0; i < K; ++i) {
                int input = sc.nextInt();
                if (input == 0) {
                    --stackSize;
                } else {
                    stack[stackSize++] = input;
                }
            }

            int answer = 0;
            while (stackSize > 0) {
                --stackSize;
                answer += stack[stackSize];
            }

            System.out.println(String.format("#%d %d", test_case, answer));
        }
    }
}
