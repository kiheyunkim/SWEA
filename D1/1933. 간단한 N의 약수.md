```c++
//C++
#include <iostream>
#include <cstring>
#include <cmath>

bool checker[1000 + 1];
int main(){
    std::cout.tie(nullptr);
    std::cin.tie(nullptr);
    std::ios::sync_with_stdio(false);
    
    memset(checker,0,sizeof(checker));
    int n;
    std::cin>>n;
    
    int sqr = sqrt(n);
    for(int i=1;i<=sqr;++i){
        if(n % i == 0){
            int num1 = i, num2 = n / i;
            if( num1 != num2){
                checker[num1] = checker[num2] = true;
            }
            else{
                checker[num1] = true;
            }
        }
    }
    
    for(int i=1;i<=n;++i){
        if(checker[i]){
            std::cout<<i << " ";
        }
    }
    std::cout<<"\n";
    
    return 0;
}
```
