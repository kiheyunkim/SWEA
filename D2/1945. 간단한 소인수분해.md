```c++
//C++
#include<iostream>
using namespace std;

int divide(int& target, int div){
    int retval = 0;
    while(target % div == 0){
        target /= div;
        ++retval;
    }
    return retval;
}

int main(int argc, char** argv){
	int T;
	cin>>T;
	for(int test_case = 1; test_case <= T; ++test_case){
        int N;
        std::cin>>N;
        int e = divide(N,11);
        int d = divide(N,7);
        int c = divide(N,5);
        int b = divide(N,3);
        int a = divide(N,2);
		std::cout<<"#"<<test_case<<" "<<a<<" "<<b<<" "<<c<<" "<<d<<" "<<e<<"\n";
    }
	
    return 0;
}
```

```java
import java.util.Scanner;

class Solution{
    public static int divide(Integer N, int div){
    	int retval = 0;
        while(N % div == 0){
        	N /= div;
            ++retval;
        }
        return retval;
    }
    
	public static void main(String args[]) throws Exception{
		Scanner sc = new Scanner(System.in);
		int T = sc.nextInt();
		for(int test_case = 1; test_case <= T; test_case++)	{
			Integer N = new Integer(sc.nextInt());
            int e = divide(N,11);
            int d = divide(N,7);
            int c = divide(N,5);
            int b = divide(N,3);
            int a = divide(N,2);
```
