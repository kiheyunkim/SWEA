//C++
#include <iostream>
#include <cstring>
int main(int argc, char* argv[]) {
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    std::ios::sync_with_stdio(false);

    int T;
    std::cin >> T;
    for (int test_case = 1; test_case <= T; ++test_case) {
        int lengthN, lengthM;
        std::cin >> lengthN >> lengthM;
        int arrayN[20], arrayM[20];
        for (int i = 0; i < lengthN; ++i) {
            std::cin >> arrayN[i];
        }
        for (int i = 0; i < lengthM; ++i) {
            std::cin >> arrayM[i];
        }

        if (lengthM > lengthN) {
            int temp[20];
            memcpy(temp, arrayN, sizeof(arrayN));
            memcpy(arrayN, arrayM, sizeof(arrayM));
            memcpy(arrayM, temp, sizeof(temp));
            
            int tempLen = lengthN;
            lengthN = lengthM;
            lengthM = tempLen;
        }

        int answer = -987654321;
        for (int i = 0; i < lengthN - lengthM + 1; ++i) {
            int sum = 0;
            for (int j = 0; j < lengthM; ++j) {
                sum += arrayN[i + j] * arrayM[j];
            }
            if (answer < sum) {
                answer = sum;
            }
        }

        std::cout << "#" << test_case << " " << answer << "\n";
    }

    return 0;
}
