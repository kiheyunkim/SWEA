#include<iostream>
#include<cstring>
int board[9][9];
bool checkCol(int col) {
    bool checker[10]{ 0, };
    for (int i = 0; i < 9; ++i) {
        if (!checker[board[col][i]]) {
            checker[board[col][i]] = true;
        }
        else {
            return false;
        }
    }
    return true;
}

bool checkRow(int row) {
    bool checker[10]{ 0, };
    for (int i = 0; i < 9; ++i) {
        if (!checker[board[i][row]]) {
            checker[board[i][row]] = true;
        }
        else {
            return false;
        }
    }
    return true;
}

bool checkSquare(int col, int row) {
    bool checker[10]{ 0, };
    for (int i = row; i < row + 3; ++i) {
        for (int j = col; j < col + 3; ++j) {
            if (!checker[board[j][i]]) {
                checker[board[j][i]] = true;
            }
            else {
                return false;
            }
        }
    }

    return true;
}


int main(int argc, char** argv) {
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    std::ios::sync_with_stdio(false);
    
    int test_case;
    int T;
    std::cin >> T;
    for (test_case = 1; test_case <= T; ++test_case) {
        for (int i = 0; i < 9; ++i) {
            for (int j = 0; j < 9; ++j) {
                std::cin >> board[j][i];
            }
        }
        bool isOk = true;
        if (isOk) {
            for (int i = 0; i < 9; ++i) {
                isOk = checkCol(i);
                if (!isOk) {
                    break;
                }
            }
        }

        if (isOk) {
            for (int i = 0; i < 9; ++i) {
                isOk = checkRow(i);
                if (!isOk) {
                    break;
                }
            }
        }

        if (isOk) {
            for (int i = 0; i < 3; ++i) {
                for (int j = 0; j < 3; ++j) {
                    isOk = checkSquare(j * 3, i * 3);
                    if (!isOk) {
                        break;
                    }
                }
                if (!isOk) {
                    break;
                }
            }
        }

        std::cout << "#" << test_case << " " << (isOk ? 1 : 0) << "\n";
    }
    return 0;//정상종료시 반드시 0을 리턴해야합니다.
}
