//C++
#include<iostream>
#include<cstring>
int main(int argc, char* argv[]) {
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    std::ios::sync_with_stdio(false);

    int T;
    std::cin >> T;
    for (int test_case = 1; test_case <= T; ++test_case) {
        int map[10][10];
        memset(map, 0, sizeof(map));
        int size, number = 1;
        std::cin >> size;
        for (int i = 0; i < size; ++i) {
            map[i][0] = number++;
        }

        std::pair<int, int> pos = std::make_pair(size - 1, 0);
        int direction = 2;
        for (int i = size - 1; i > 0; --i) {
            for (int j = 0; j < 2; ++j) {
                for (int k = 0; k <= i - 1; ++k) {
                    if (direction == 0) {
                        --pos.second;
                    }
                    else if (direction == 1) {
                        ++pos.first;
                    }
                    else if (direction == 2) {
                        ++pos.second;
                    }
                    else if (direction == 3) {
                        --pos.first;
                    }
                    map[pos.first][pos.second] = number++;
                }
                direction = (direction + 1) % 4;
            }    
        }

        std::cout << "#" << test_case << "\n";
        for (int i = 0; i < size; ++i) {
            for (int j = 0; j < size; ++j) {
                std::cout << map[j][i] << " ";
            }
            std::cout << "\n";
        }

        std::cout << "\n";
    }

    return 0;
}
