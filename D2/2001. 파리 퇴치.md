//C++
#include<iostream>
int board[15][15];
int check(std::pair<int,int> pos, int size){
    int retval = 0;
    for(int i=pos.second;i<pos.second+size;++i){
        for(int j=pos.first;j<pos.first+size;++j){
            retval += board[j][i];
    	}
    }
    
    return retval;
}

int main(int argc, char** argv){
    std::cout.tie(nullptr);
    std::cin.tie(nullptr);
    std::ios::sync_with_stdio(false);
	
    int T;
	std::cin>>T;
	for(int test_case = 1; test_case <= T; ++test_case){
        int n,m;
        std::cin>>n>>m;
        for(int i=0;i<n;++i){
        	for(int j=0;j<n;++j){
                std::cin>>board[j][i];
            }
        }
        
        int answer = -987654321;
        for(int i=0;i<n-m+1;++i){
            for(int j=0;j<n-m+1;++j){
                int retval = check(std::make_pair(j,i),m);
                if(answer < retval){
                    answer = retval;
                }
            }
        }
        
        std::cout<<"#"<<test_case<<" "<<answer<<"\n";
	}
	return 0;//정상종료시 반드시 0을 리턴해야합니다.
}

//java
import java.util.Scanner;
import java.io.FileInputStream;
class Solution{
    public static int check(int[][] board, int startX, int startY, int size){
        int retval = 0;
        for(int i=startY;i<startY+size;++i){
            for(int j=startX;j<startX+size;++j){
                retval += board[j][i];
            }
        }
        return retval;
    }
    
	public static void main(String args[]) throws Exception{
		Scanner sc = new Scanner(System.in);
		int T=sc.nextInt();
		for(int test_case = 1; test_case <= T; test_case++)	{
			int[][] board = new int[15][15];
            int n = sc.nextInt(), m = sc.nextInt();
            for(int i=0;i<n;++i){
                for(int j=0;j<n;++j){
                    board[j][i] = sc.nextInt();
                }
            }
            
            int answer = -987654321;
            for(int i=0;i<n-m+1;i++){
                for(int j=0;j<n-m+1;j++){
                    int retval = check(board,j,i,m);
                    if(answer < retval){
                        answer = retval;
                    }
            	}
            }
            
            System.out.println(String.format("#%d %d",test_case, answer));
		}
	}
}
