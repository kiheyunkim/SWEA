#include <iostream>

int days(int month){
	switch(month){
        case 1:
        case 3:
        case 5:
        case 7:
        case 8:
        case 10:
        case 12:
            return 31;
        case 2:
            return 28;
        case 4:
        case 6:
        case 9:
        case 11:
            return 30;
    }
    
    return -1;
}

int main(int argc, char* argv[]){
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
    std::ios::sync_with_stdio(false);
    
    int T;
    std::cin>>T;
    for(int test_case = 1; test_case <= T; ++test_case){
    	int m1,d1,m2,d2;
        std::cin>>m1>>d1>>m2>>d2;
        int day1 = 0, day2 = 0;
        for(int i=1;i<m1;++i){
            day1 += days(i);
        }
        day1 += d1;
        
        for(int i=1;i<m2;++i){
            day2 += days(i);
        }
        day2 += d2;
        
        std::cout<<"#"<<test_case<<" "<<day2 - day1 + 1<<"\n";
    }
    
	return 0;
}

                return 31;

	public static void main(String args[]) throws Exception{
